212void IWAPP::InitMenuCmd(HMENU hmenu, int cmd, const unique_ptr<ICMD>& pcmd)
213{
214 MENUITEMINFOW mi = { sizeof(mi) };
215 mi.fMask = MIIM_STATE;
216 mi.fState = pcmd->FEnabled() ? (UINT)MFS_UNCHECKED|MFS_ENABLED : (UINT)MFS_UNCHECKED|MFS_DISABLED|MF_GRAYED;
217 if (pcmd->FChecked())
218 mi.fState |= MFS_CHECKED;
219 string sMenu;
220 wstring wsMenu; // needs scope so the underlying string doesn't get freed until we use it
221 if (pcmd->FMenuS(sMenu)) {
222 mi.fMask |= MIIM_TYPE;
223 wsMenu = WsFromS(sMenu);
224 mi.dwTypeData = const_cast<LPWSTR>(wsMenu.c_str());
225 }
226 ::SetMenuItemInfoW(hmenu, (UINT)cmd, false, &mi);
227}

240bool IWAPP::FVerifySubMenuCmdsRegistered(HMENU hmenu) const
241{
242 MENU menu(hmenu);
243 for (MENUITEMINFOW mii : menu) {
244 if (mii.wID == 0)
245 continue;
246 if (mii.hSubMenu) {
247 if (!FVerifySubMenuCmdsRegistered(mii.hSubMenu))
248 return false;
249 }
250 else if (mpcmdpicmdMenu.find((int)mii.wID) == mpcmdpicmdMenu.end())
251 return false;
252 }
253 return true;
254}
