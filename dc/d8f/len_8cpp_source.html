102void LEN::PositionLeft(WN& wn, const SZ& sz)
103{
104 /* if we're aleady beyond the right edge, go ahead and wrap now */
105 if (rcFlow.left >= rcFlow.right)
106 rcFlow = RC(rcWithin.left, rcFlow.bottom, rcWithin.right, rcFlow.bottom);
107
108 /* layout the control within the flow area */
109 RC rc(rcFlow);
110 rc.SetSz(sz);
111
112 /* if we're beyond the right edge now, wrap and re-layout with extra width */
113 if (rc.right > rcFlow.right) {
114 rcFlow = RC(rcWithin.left, rcFlow.bottom, rcWithin.right, rcFlow.bottom);
115 rc = rcFlow;
116 rc.SetSz(wn.SzIntrinsic(rc.RcSetBottom(rcWithin.bottom)));
117 }
118
119 /* position the control */
120 wn.SetBounds(rc);
121
122 /* adjust the flow area */
123 rcFlow.left = rc.right + marginDef.right;
124 rcFlow.bottom = max(rc.bottom, rcFlow.bottom);
125}

197{
198 PT ptCenterEnd(rcWithin.ptTopLeft());
199 SZ sz;
200
201 switch (cen) {
202 case CEN::Vertical:
203 ptCenterEnd.y -= marginDef.bottom;
204 sz = SZ(0, (szCenterTotal.height - (ptCenterEnd.y - ptCenterStart.y))/2);
205 for (WN* pwn : vpwn)
206 pwn->SetBounds(pwn->RcBounds() + sz);
207 break;
208 case CEN::Horizontal:

213 break;

272void LE::Position(void) noexcept
273{
274 RC rcWithin = wnContainer.RcInterior();
275 rcWithin.Unpad(margin);
276
277 for (auto ppwnChild = wnContainer.vpwnChildren.begin(); ppwnChild != wnContainer.vpwnChildren.end(); ppwnChild++) {
278 RC rcNew = rcWithin;
279 rcNew.SetWidth(mppwnrc[*ppwnChild].dxWidth());
280 LEIT leit = (*ppwnChild)->Leit();
281 if (leit.lealignh == LEALIGNH::Left) {
282 rcWithin.left = rcNew.right + gutter.width;
283 }
284 else if (leit.lealignh == LEALIGNH::Right) {
285 rcNew += SZ(rcWithin.right - rcNew.right, 0);
286 rcWithin.right = rcNew.left - gutter.width;
287 }
288 AlignV(rcNew, rcWithin, leit.lealignv);
289 mppwnrc[*ppwnChild] = rcNew;
290 }

304void LE::AlignV(RC& rcItem, const RC& rcWithin, LEALIGNV lealignv) noexcept
305{
306 float dyItem = rcItem.dyHeight();
