146class HSV
147{
148public:
149 HSV(float hue, float sat, float val) : hue(hue), sat(sat), val(val) {}
150
151 HSV(CO co)
152 {
153 float wMax = max(co.r, max(co.g, co.b));
154 float wMin = min(co.r, min(co.g, co.b));
155 val = wMax;
156 if (wMax == 0)
157 sat = hue = 0;
158 else {
159 float dw = wMax - wMin;
160 sat = dw / wMax;
161 if (co.r == wMax)
162 hue = (co.g - co.b) / wMax;
163 else if (co.g == wMax)
164 hue = 2 + (co.b - co.r) / wMax;
165 else
166 hue = 4 + (co.r - co.g) / wMax;
167 hue *= 60;
168 if (hue < 0)
169 hue += 360;
170 }
171 }
172
173 operator CO() const
174 {
175 int sex = (int)(hue / 60) % 6;
176 float dsex = (hue / 60) - (float)sex; // distnce from the sextant boundary
177 float p = val * (1 - sat);
178 float q = val * (1 - dsex*sat);
179 float t = val * (1 - (1-dsex)*sat);
180
181 switch (sex) {
182 default:
183 case 0: return CO(val, t, p); // 0-60
184 case 1: return CO(q, val, p); // 60-120
185 case 2: return CO(p, val, t); // 120-180
186 case 3: return CO(p, q, val); // 180-240
187 case 4: return CO(t, p, val); // 240-300
188 case 5: return CO(val, p, q); // 300-360
189 }
190 }
191
192 HSV& SetHue(float hueNew)
193 {
194 hue = hueNew;
195 return *this;

200 sat = satNew;

213 return *this;
214 }

216public:

262inline float CompBlend(float a, float b, float alpha)
263{
264 return sqrt((1.f-alpha) * a*a +
265 alpha * b*b);
266}
267
268inline CO CoBlend(CO co1, CO co2, float pct = 0.5f)
269{
270 return CO(CompBlend(co1.r, co2.r, pct),
271 CompBlend(co1.g, co2.g, pct),
272 CompBlend(co1.b, co2.b, pct),
273 (1.f-pct) * co1.a + pct * co2.a); /* alpha blends linearly */
274}
