74void DCP::SetFont(TF& tf, const string& sFace, float dyHeight, TF::WEIGHT weight, TF::STYLE style)
75{
76 if (tf.hfont) {
77 ::DeleteObject(tf.hfont);
78 tf.hfont = NULL;
79 }
80 LOGFONTW lf = { 0 };
81 lf.lfHeight = -(int)roundf(dyHeight);
82 lf.lfWidth = 0;
83 lf.lfEscapement = 0;
84 lf.lfOrientation = 0;
85 lf.lfWeight = mpweightfw[(int)weight];
86 lf.lfItalic = style == TF::STYLE::Italic;
87 lf.lfUnderline = 0;
88 lf.lfStrikeOut = 0;
89 lf.lfCharSet = DEFAULT_CHARSET;
90 lf.lfOutPrecision = OUT_DEFAULT_PRECIS;
91 lf.lfClipPrecision = CLIP_DEFAULT_PRECIS;
92 lf.lfPitchAndFamily = DEFAULT_PITCH | FF_DONTCARE;
93 lstrcpyW(lf.lfFaceName, WsFromS(sFace).c_str());
94 tf.hfont = ::CreateFontIndirectW(&lf);
95}

115void DCP::SetFontWidth(TF& tf, float dxWidth)
116{
117 assert(tf.hfont != NULL);
118 HGDIOBJ hfontSav = ::SelectObject(hdc, tf.hfont);
119 TEXTMETRICW tm;
120 ::GetTextMetricsW(hdc, &tm);
121 ::SelectObject(hdc, hfontSav);
122 LOGFONTW lf;
123 ::GetObject(tf.hfont, sizeof(lf), &lf);
124 lf.lfHeight = -(int)roundf((float)(tm.tmHeight-tm.tmInternalLeading) * dxWidth / (float)tm.tmAveCharWidth);
125 lf.lfWidth = 0;// (int)roundf(dxWidth);
126 ::DeleteObject(tf.hfont);
127 tf.hfont = NULL;
128 tf.hfont = ::CreateFontIndirectW(&lf);
129}

193SZ DCP::SzFromS(const string& s, const TF& tf, float dxWidth) const
194{
195 if (dxWidth < 0)
196 dxWidth = 32767.0f;
197 wstring ws = WsFromS(s);
198 RECT rect = { 0, 0, (int)roundf(dxWidth), 0 };
199 HGDIOBJ hfontSav = ::SelectObject(hdc, tf);
200 SZ sz(0);
201 if (ws.length() == 0) {
202 TEXTMETRICW tm;
203 ::GetTextMetrics(hdc, &tm);
204 sz = SZ((float)0, (float)tm.tmHeight);
205 }
206 else {
207 SIZE size;
208 ::GetTextExtentPoint32W(hdc, ws.c_str(), (int)ws.length(), &size);
209 if (size.cx <= rect.right - rect.left)
210 sz = size;
211 else {
212 ::DrawTextW(hdc, ws.c_str(), (int)ws.length(), &rect, DT_WORDBREAK | DT_CALCRECT | DT_NOPREFIX);
213 sz = SZ(dxWidth, (float)(rect.bottom - rect.top));
214 }
215 }
216 ::SelectObject(hdc, hfontSav);
217 return sz;
218}

220FM DCP::FmFromTf(const TF& tf) const
221{
222 FM fm(0);
223 HGDIOBJ hfontSav = ::SelectObject(hdc, tf);
224 TEXTMETRICW tm;
225 ::GetTextMetrics(hdc, &tm);
226 ::SelectObject(hdc, hfontSav);
227
228 fm.dyAscent = (float)tm.tmAscent;
229 fm.dyDescent = (float)tm.tmDescent;
230 fm.dyXHeight = (float)3*tm.tmAscent/4;
231 fm.dyCapHeight = (float)(tm.tmAscent - tm.tmInternalLeading);
232
233 fm.dyLineGap = (float)tm.tmExternalLeading;
234
235 return fm;
236}
