296void WN::RedrawRcg(RC rcgUpdate, DRO dro)
297{
298 if (!FVisible())
299 return;
300 /* Present's rectangle is an integer-based RECT, not a Direct2D float rectangle. That
301 means we must expand the update rectangle to its encompassing integer boundaries.
302 And that means potentially redrawing this WN's parent to fill those partial pixels.
303 This is potentially very expensive, so we do this only when we absolutely must.
304 Applications that care about performance should integer-align WNs */
305 if (rcgUpdate.FRoundUp()) {
306 assert(pwnParent); // the top-most WN is guaranteed to be integer-aligned
307 pwnParent->RedrawRcg(rcgUpdate, dro);
308 }
309 else {
310 if (!FBeginDraw()) {
311 /* buffer is dirty - force full redraw of the top-level window */
312 rcgUpdate = iwapp.RcInterior();
313 iwapp.DrawWithChildren(rcgUpdate, droParentNotDrawn);
314 }
315 else {
316 DrawWithChildren(rcgUpdate, dro);
317 DrawOverlappedSiblings(rcgUpdate);
318 }
319 EndDraw(rcgUpdate);
320 }
321}

335void WN::DrawNoChildren(const RC& rcgUpdate, DRO dro)
336{
337 assert(fVisible);
338 iwapp.prt->PushAxisAlignedClip(rcgUpdate, D2D1_ANTIALIAS_MODE_PER_PRIMITIVE);
339
340 RC rcDraw = RcFromRcg(rcgUpdate);
341 Erase(rcDraw, dro);
342 Draw(rcDraw);

344 iwapp.prt->PopAxisAlignedClip();
