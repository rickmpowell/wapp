2/*
3 * dlg.cpp
4 *
5 * Dialog boxes.
6 *
7 * Dialogs work by taking input parameters in a structure, which is used to
8 * populate the dialog controls, and returns the same structure filled in
9 * with updated values.
10 *
11 * Error checking is performed prior to dismissing a dialog and the dialog
12 * is not dismissed until the errors are clear. This implies it's possible
13 * for the dialog to contain values that are not legal during intermediate
14 * stages, so errors need to detected and handled cleanly, and "illegal"
15 * states are not unusual in normal operation.
16 *
17 * Best practices:
18 * Implement custom controls that represent the types/classes in the
19 * application.
20 * The custom controls include decoders, parsers, error detection,
21 * and may hold state in both raw and parsed (i.e., typed) formats
22 * In cases where multiple controls need consistency between them,
23 * (i.e., error checking is not contained inside a single control
24 * itself), consistency checking should be performed in the dialog
25 * box, not the controls.
26 *
27 * Controls should implement
28 * SetData - takes application-specific data type
29 * ErrParseData - parses the raw data into app-specific type, reports
30 * errors on failures
31 * DataGet - Returns the applicatoin-specific data
32 * These are not virtual function, since the types are app-specific.
33 * Naming is just a convention.
34 *
35 * Dialogs should implement
36 * constructor with the object we're operating on as a parameter
37 * ErrValidate - validation routine
38 * ExtractData - moves the data from the dialog into the app object
39 */

48static INT_PTR CALLBACK DlgProc(HWND hwnd, UINT wm, WPARAM wParam, LPARAM lParam)
49{
50 switch (wm) {
51
52 case WM_INITDIALOG:
53 return (INT_PTR)TRUE;
54
55 case WM_COMMAND:
56 switch (LOWORD(wParam)) {
57 case IDOK:
58 case IDCANCEL:
59 ::EndDialog(hwnd, LOWORD(wParam));
60 return (INT_PTR)TRUE;
61 }
62 break;
63 }
64
65 return (INT_PTR)FALSE;
66}

187void DLGFILE::BuildFilter(wchar_t* wsFilter, int cchFilter)
188{
189 wchar_t* pchFilter = wsFilter;
190 for (auto it : mpextsLabel) {
191 wcsncpy_s(pchFilter, cchFilter - (pchFilter - wsFilter) - 3, WsFromS(it.second).c_str(), _TRUNCATE); // second has label
192 pchFilter += wcslen(pchFilter);
193 *pchFilter++ = 0;
194 *pchFilter++ = L'*';
195 *pchFilter++ = L'.';
196 wcsncpy_s(pchFilter, cchFilter - (pchFilter - wsFilter) - 3, WsFromS(it.first).c_str(), _TRUNCATE);
197 pchFilter += wcslen(pchFilter);
198 *pchFilter++ = 0;
199 }
200 *pchFilter++ = 0; // double null-terminate the final filter item
201}

208OFN DLGFILE::OfnDefault(void)
209{
210 OFN ofn(1024);
211 ofn.ofn.hwndOwner = iwapp.hwnd;
212 ofn.ofn.hInstance = iwapp.hinst;
213 BuildFilter(ofn.wsFilter.get(), 1024);
214 ofn.ofn.lpstrFilter = ofn.wsFilter.get();
215 auto it = mpextsLabel.find(extDefault);
216 ofn.ofn.nFilterIndex = (int)distance(mpextsLabel.begin(), it) + 1;
217 ofn.ofn.lpstrFile = ofn.wsFile.get();
218 ofn.ofn.nMaxFile = 1024;
219 ofn.ofn.lpstrDefExt = WsFromS(extDefault).c_str();
220 ofn.ofn.lpstrInitialDir = ofn.wsDirectory.get();
221
222 if (!file.empty()) {
223 wcscpy_s(ofn.wsFile.get(), 1024, WsFromS(file.filename().string()).c_str());
224 wcscpy_s(ofn.wsDirectory.get(), 1024, WsFromS(file.parent_path().string()).c_str());
225 }
226
227 return move(ofn);
228}

295bool DLGFOLDER::FRun(void)
296{
297 com_ptr<IFileDialog> pdlg;
298 ThrowError(CoCreateInstance(CLSID_FileOpenDialog,
299 NULL,
300 CLSCTX_INPROC_SERVER,
301 __uuidof(pdlg),
302 &pdlg));
303 DWORD opt;
304 ThrowError(pdlg->GetOptions(&opt));
305 ThrowError(pdlg->SetOptions(opt| FOS_PICKFOLDERS));
306 HRESULT hr = pdlg->Show(NULL);
307 if (!SUCCEEDED(hr))
308 return false;
309 com_ptr<IShellItem> psi;
310 pdlg->GetResult(&psi);
311 PWSTR wsPath = nullptr;
312 psi->GetDisplayName(SIGDN_FILESYSPATH, &wsPath);
313 folder = SFromWs(wstring_view(wsPath));
314 CoTaskMemFree(wsPath);
315 return true;
316}

327bool DLGPRINT::FRun(void)
328{
329 PRINTDLGEXW pd = { sizeof(pd) };
330 pd.hwndOwner = iwapp.hwnd;
331 pd.Flags = PD_RETURNDC | PD_NOPAGENUMS;
332 pd.nCopies = 1;
333 pd.nStartPage = START_PAGE_GENERAL;
334 pd.hDevNames = NULL;
335
336 global_ptr<DEVMODEW> pdevmode(1, GMEM_MOVEABLE | GMEM_ZEROINIT);
337 pd.hDevMode = pdevmode.handle();
338 pdevmode->dmSize = sizeof(DEVMODEW);
339 pdevmode->dmFields = DM_ORIENTATION | DM_PAPERSIZE;
340 pdevmode->dmOrientation = DMORIENT_LANDSCAPE;
341 pdevmode->dmPaperSize = DMPAPER_LETTER;
342 pdevmode.unlock();
343
344 HRESULT hr = ::PrintDlgEx(&pd);
345 if (!SUCCEEDED(hr) || pd.dwResultAction != PD_RESULT_PRINT || pd.hDC == NULL)
346 return false;
347
348 hdc = pd.hDC;
349 return true;
350}
