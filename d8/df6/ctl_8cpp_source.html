592void TOOLBAR::Measure(void) noexcept
593{
594 RC rcWithin = RcInterior();
595 float dxMargin = rcWithin.dyHeight() / 4;
596 margin = PAD(dxMargin, 1, dxMargin, 2);
597 gutter = SZ(2*dxMargin, 0);
598 rcWithin.Unpad(margin);
599
600 for (auto ppwnChild = vpwnChildren.begin(); ppwnChild != vpwnChildren.end(); ppwnChild++) {
601 SZ szChild = (*ppwnChild)->SzIntrinsic(RcInterior());
602 /* scale up to fit vertically and save for locate phase */
603 LEIT leit = (*ppwnChild)->Leit();
604 if (leit.lestretch == LESTRETCH::KeepWidth)
605 szChild.height = rcWithin.dyHeight();
606 else if (leit.lestretch == LESTRETCH::KeepAspect)
607 szChild *= rcWithin.dyHeight() / szChild.height;
608 mppwnrc[*ppwnChild] = RC(PT(0), szChild);
609 }
