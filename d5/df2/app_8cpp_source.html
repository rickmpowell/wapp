300LRESULT CALLBACK WND::WndProc(HWND hwnd, UINT wm, WPARAM wParam, LPARAM lParam)
301{
302 /* TODO: we should probably catch exceptions here and display an error message */
303
304 /* messages that come in before we have the HWND and WN pointing at one another
305 need default handling. link the HWND and WN during WM_NCCREATE. */
306
307 WND* pwnd = (WND*)GetWindowLongPtrW(hwnd, GWLP_USERDATA);
308 if (pwnd == nullptr) {
309 if (wm != WM_NCCREATE)
310 return ::DefWindowProcW(hwnd, wm, wParam, lParam);
311 pwnd = (WND*)((CREATESTRUCT*)lParam)->lpCreateParams;
312 pwnd->hwnd = hwnd;
313 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, (LONG_PTR)pwnd);
314 }
315
316 /* dispatch window messages */
317
318 switch (wm) {
319
320 case WM_CREATE:
321 pwnd->OnCreate();
322 break;
323
324 case WM_DESTROY:
325 pwnd->OnDestroy();
326 pwnd->hwnd = NULL;
327 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, NULL);
328 return 0;
329
330 case WM_SIZE:
331 switch (wParam) {
332 case SIZE_MINIMIZED:
333 case SIZE_RESTORED:
334 pwnd->OnMinimize(wParam == SIZE_MINIMIZED);
335 break;
336 default:
337 break;
338 }
339 pwnd->OnSize(SZ(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)));
340 InvalidateRect(hwnd, NULL, FALSE);
341 break;
342
343 case WM_SHOWWINDOW:
344 pwnd->OnShow((bool)wParam);
345 break;
346
347 case WM_ERASEBKGND:
348 return TRUE;
349
350 case WM_PAINT:
351 pwnd->OnPaint();
352 return 0;
353
354 case WM_DISPLAYCHANGE:
355 pwnd->OnDisplayChange();
356 return 0;
357
358 case WM_MOUSEMOVE:
359 pwnd->OnMouseMove(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), (unsigned)wParam);
360 return 0;
361
362 case WM_LBUTTONDOWN:
363 pwnd->OnMouseDown(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
364 return 0;
365
366 case WM_LBUTTONUP:
367 pwnd->OnMouseUp(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
368 return 0;
369
370 case WM_MOUSEWHEEL:
371 {
372 POINT pt = { GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam) };
373 ::ScreenToClient(hwnd, &pt);
374 pwnd->OnMouseWheel(pt, GET_WHEEL_DELTA_WPARAM(wParam));
375 return 0;
376 }
377
378 case WM_KEYDOWN:
379 pwnd->OnKeyDown((int)wParam);
380 return 0;
381
382 case WM_TIMER:
383 pwnd->OnTimer((int)wParam);
384 return 0;
385
386 case WM_COMMAND:
387 if (pwnd->OnCommand(LOWORD(wParam)))
388 return 0;
389 break;
390
391 case WM_INITMENU:
392 pwnd->OnInitMenu();
393 return 0;
394
395 case WM_INITMENUPOPUP:
396 pwnd->OnInitMenuPopup((HMENU)wParam);
397 return 0;
398
399 default:
400 break;
401 }
402
403 return pwnd->DefProc(wm, wParam, lParam);
404}
