294LRESULT CALLBACK WND::WndProc(HWND hwnd, UINT wm, WPARAM wParam, LPARAM lParam)
295{
296 /* TODO: we should probably catch exceptions here and display an error message */
297
298 /* messages that come in before we have the HWND and WN pointing at one another
299 need default handling. link the HWND and WN during WM_NCCREATE. */
300
301 WND* pwnd = (WND*)GetWindowLongPtrW(hwnd, GWLP_USERDATA);
302 if (pwnd == nullptr) {
303 if (wm != WM_NCCREATE)
304 return ::DefWindowProcW(hwnd, wm, wParam, lParam);
305 pwnd = (WND*)((CREATESTRUCT*)lParam)->lpCreateParams;
306 pwnd->hwnd = hwnd;
307 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, (LONG_PTR)pwnd);
308 }
309
310 /* dispatch window messages */
311
312 switch (wm) {
313
314 case WM_CREATE:
315 pwnd->OnCreate();
316 break;
317
318 case WM_DESTROY:
319 pwnd->OnDestroy();
320 pwnd->hwnd = NULL;
321 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, NULL);
322 return 0;
323
324 case WM_SIZE:
325 switch (wParam) {
326 case SIZE_MINIMIZED:
327 case SIZE_RESTORED:
328 pwnd->OnMinimize(wParam == SIZE_MINIMIZED);
329 break;
330 default:
331 break;
332 }
333 pwnd->OnSize(SZ(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)));
334 InvalidateRect(hwnd, NULL, FALSE);
335 break;
336
337 case WM_SHOWWINDOW:
338 pwnd->OnShow((bool)wParam);
339 break;
340
341 case WM_ERASEBKGND:
342 return TRUE;
343
344 case WM_PAINT:
345 pwnd->OnPaint();
346 return 0;
347
348 case WM_DISPLAYCHANGE:
349 pwnd->OnDisplayChange();
350 return 0;
351
352 case WM_MOUSEMOVE:
353 pwnd->OnMouseMove(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), (unsigned)wParam);
354 return 0;
355
356 case WM_LBUTTONDOWN:
357 pwnd->OnMouseDown(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
358 return 0;
359
360 case WM_LBUTTONUP:
361 pwnd->OnMouseUp(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
362 return 0;
363
364 case WM_MOUSEWHEEL:
365 {
366 POINT pt = { GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam) };
367 ::ScreenToClient(hwnd, &pt);
368 pwnd->OnMouseWheel(pt, GET_WHEEL_DELTA_WPARAM(wParam));
369 return 0;
370 }
371
372 case WM_KEYDOWN:
373 pwnd->OnKeyDown((int)wParam);
374 return 0;
375
376 case WM_TIMER:
377 pwnd->OnTimer((int)wParam);
378 return 0;
379
380 case WM_COMMAND:
381 if (pwnd->OnCommand(LOWORD(wParam)))
382 return 0;
383 break;
384
385 case WM_INITMENU:
386 pwnd->OnInitMenu();
387 return 0;
388
389 case WM_INITMENUPOPUP:
390 pwnd->OnInitMenuPopup((HMENU)wParam);
391 return 0;
392
393 default:
394 break;
395 }
396
397 return pwnd->DefProc(wm, wParam, lParam);
398}
