240LRESULT CALLBACK WND::WndProc(HWND hwnd, UINT wm, WPARAM wParam, LPARAM lParam)
241{
242 /* messages that come in before we have the HWND and WN pointing at one another
243 need default handling. link the HWND and WN during WM_NCCREATE. */
244
245 WND* pwnd = (WND*)GetWindowLongPtrW(hwnd, GWLP_USERDATA);
246 if (pwnd == nullptr) {
247 if (wm != WM_NCCREATE)
248 return ::DefWindowProcW(hwnd, wm, wParam, lParam);
249 pwnd = (WND*)((CREATESTRUCT*)lParam)->lpCreateParams;
250 pwnd->hwnd = hwnd;
251 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, (LONG_PTR)pwnd);
252 }
253
254 /* dispatch window messages */
255
256 switch (wm) {
257
258 case WM_CREATE:
259 pwnd->OnCreate();
260 break;
261
262 case WM_DESTROY:
263 pwnd->OnDestroy();
264 pwnd->hwnd = NULL;
265 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, NULL);
266 return 0;
267
268 case WM_SIZE:
269 pwnd->OnSize(SZ(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)));
270 InvalidateRect(hwnd, NULL, FALSE);
271 break;
272
273 case WM_SHOWWINDOW:
274 pwnd->OnShow((bool)wParam);
275 break;
276
277 case WM_ERASEBKGND:
278 return TRUE;
279
280 case WM_PAINT:
281 pwnd->OnPaint();
282 return 0;
283
284 case WM_DISPLAYCHANGE:
285 pwnd->OnDisplayChange();
286 return 0;
287
288 case WM_MOUSEMOVE:
289 pwnd->OnMouseMove(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), (unsigned)wParam);
290 return 0;
291
292 case WM_LBUTTONDOWN:
293 pwnd->OnMouseDown(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
294 return 0;
295
296 case WM_LBUTTONUP:
297 pwnd->OnMouseUp(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
298 return 0;
299
300 case WM_MOUSEWHEEL:
301 {
302 POINT pt = { GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam) };
303 ::ScreenToClient(hwnd, &pt);
304 pwnd->OnMouseWheel(pt, GET_WHEEL_DELTA_WPARAM(wParam));
305 return 0;
306 }
307
308 case WM_KEYDOWN:
309 pwnd->OnKeyDown((int)wParam);
310 return 0;
311
312 case WM_COMMAND:
313 if (pwnd->OnCommand(LOWORD(wParam)))
314 return 0;
315 break;
316
317 case WM_INITMENU:
318 pwnd->OnInitMenu();
319 return 0;
320
321 case WM_INITMENUPOPUP:
322 pwnd->OnInitMenuPopup((HMENU)wParam);
323 return 0;
324
325 default:
326 break;
327 }
328
329 return pwnd->DefProc(wm, wParam, lParam);
330}
