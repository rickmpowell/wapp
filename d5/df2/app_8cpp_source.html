353LRESULT CALLBACK WND::WndProc(HWND hwnd, UINT wm, WPARAM wParam, LPARAM lParam)
354{
355 /* TODO: we should probably catch exceptions here and display an error message */
356
357 /* messages that come in before we have the HWND and WN pointing at one another
358 need default handling. link the HWND and WN during WM_NCCREATE. */
359
360 WND* pwnd = (WND*)GetWindowLongPtrW(hwnd, GWLP_USERDATA);
361 if (pwnd == nullptr) {
362 if (wm != WM_NCCREATE)
363 return ::DefWindowProcW(hwnd, wm, wParam, lParam);
364 pwnd = (WND*)((CREATESTRUCT*)lParam)->lpCreateParams;
365 pwnd->hwnd = hwnd;
366 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, (LONG_PTR)pwnd);
367 }
368
369 /* dispatch window messages */
370
371 switch (wm) {
372
373 case WM_CREATE:
374 pwnd->OnCreate();
375 break;
376
377 case WM_DESTROY:
378 pwnd->OnDestroy();
379 pwnd->hwnd = NULL;
380 ::SetWindowLongPtrW(hwnd, GWLP_USERDATA, NULL);
381 return 0;
382
383 case WM_SIZE:
384 switch (wParam) {
385 case SIZE_MINIMIZED:
386 case SIZE_RESTORED:
387 pwnd->OnMinimize(wParam == SIZE_MINIMIZED);
388 break;
389 default:
390 break;
391 }
392 pwnd->OnSize(SZ(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)));
393 InvalidateRect(hwnd, NULL, FALSE);
394 break;
395
396 case WM_SHOWWINDOW:
397 pwnd->OnShow((bool)wParam);
398 break;
399
400 case WM_ERASEBKGND:
401 return TRUE;
402
403 case WM_PAINT:
404 pwnd->OnPaint();
405 return 0;
406
407 case WM_DISPLAYCHANGE:
408 pwnd->OnDisplayChange();
409 return 0;
410
411 case WM_MOUSEMOVE:
412 pwnd->OnMouseMove(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), (unsigned)wParam);
413 return 0;
414
415 case WM_LBUTTONDOWN:
416 pwnd->OnMouseDown(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
417 return 0;
418
419 case WM_LBUTTONUP:
420 pwnd->OnMouseUp(PT(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam)), MK_LBUTTON);
421 return 0;
422
423 case WM_MOUSEWHEEL:
424 {
425 POINT pt = { GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam) };
426 ::ScreenToClient(hwnd, &pt);
427 pwnd->OnMouseWheel(pt, GET_WHEEL_DELTA_WPARAM(wParam));
428 return 0;
429 }
430
431 case WM_KEYDOWN:
432 pwnd->OnKeyDown((int)wParam);
433 return 0;
434
435 case WM_TIMER:
436 pwnd->OnTimer((int)wParam);
437 return 0;
438
439 case WM_COMMAND:
440 if (pwnd->OnCommand(LOWORD(wParam)))
441 return 0;
442 break;
443
444 case WM_INITMENU:
445 pwnd->OnInitMenu();
446 return 0;
447
448 case WM_INITMENUPOPUP:
449 pwnd->OnInitMenuPopup((HMENU)wParam);
450 return 0;
451
452 default:
453 break;
454 }
455
456 return pwnd->DefProc(wm, wParam, lParam);
457}
