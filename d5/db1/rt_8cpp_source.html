14 * state saved to rebuild themselves from their internally saved state and the new

18vector<DDDO*>* RTC::pvpdddo;

20void RTC::RegisterDevDeps(DDDO& dddo)
21{
22 /* NOTE: we do not take ownership of the dddo here */
23 if (!pvpdddo)
24 pvpdddo = new vector<DDDO*>;
25 pvpdddo->push_back(&dddo);
26}
27
28void RTC::UnregisterDevDeps(DDDO& dddo)
29 {
30 assert(pvpdddo);
31
32 auto ipdddo = std::find(pvpdddo->begin(), pvpdddo->end(), &dddo);
33 if (ipdddo != pvpdddo->end())
34 pvpdddo->erase(ipdddo);
35
36 if (pvpdddo->size() == 0) {
37 delete pvpdddo;
38 pvpdddo = nullptr;
39 }
40}
41
42void RTC::PurgeRegisteredDevDeps(void)
43{
44 if (!pvpdddo)
45 return;
46 for (DDDO* pdddo: *pvpdddo)
47 pdddo->purge();
48}
49
50void RTC::RebuildRegisteredDevDeps(IWAPP& iwapp)
51{
52 if (!pvpdddo)
53 return;
54 for (DDDO* pdddo : *pvpdddo)
55 pdddo->rebuild(iwapp);
56}

70void RTCFLIP::RebuildDev()
71{
72 /* get the Direct3D 11 device and device context */
73
74 D3D_FEATURE_LEVEL afld3[] = {
75 D3D_FEATURE_LEVEL_11_1, D3D_FEATURE_LEVEL_11_0,
76 D3D_FEATURE_LEVEL_10_1, D3D_FEATURE_LEVEL_10_0,
77 D3D_FEATURE_LEVEL_9_3, D3D_FEATURE_LEVEL_9_2, D3D_FEATURE_LEVEL_9_1
78 };
79 com_ptr<ID3D11Device> pdev3T;
80 com_ptr<ID3D11DeviceContext> pdc3T;
81 D3D_FEATURE_LEVEL level3 = D3D_FEATURE_LEVEL_1_0_CORE;
82 ThrowError(D3D11CreateDevice(nullptr,
83 D3D_DRIVER_TYPE_HARDWARE,
84 nullptr,
85#ifndef NDEBUG
86 D3D11_CREATE_DEVICE_BGRA_SUPPORT | D3D11_CREATE_DEVICE_DEBUG,
87#else
88 D3D11_CREATE_DEVICE_BGRA_SUPPORT,
89#endif
90 afld3, size(afld3),
91 D3D11_SDK_VERSION,
92 &pdev3T, &level3, &pdc3T));
93 pdev3T.As(&pdev3);
94 pdc3T.As(&pdc3);
95
96 /* create the Direct2D device */
97
98 pdev3.As(&pdevxgi);
99 ThrowError(iwapp.pfactd2->CreateDevice(pdevxgi.Get(), &pdev2));
100
101 /* and get the DirectX Graphics interface factory, which is used to create the
102 swap chain and back buffer */
103
104 com_ptr<IDXGIAdapter> padaptxgiT;
105 pdevxgi->GetAdapter(&padaptxgiT);
106 padaptxgiT->GetParent(IID_PPV_ARGS(&pfactxgi));
107}

125void RTCFLIP::RebuildDevDeps(com_ptr<ID2D1DeviceContext>& pdc2)
126{
127 if (pdc2)
128 return;
129
130 RebuildDev();
131
132 ThrowError(pdev2->CreateDeviceContext(D2D1_DEVICE_CONTEXT_OPTIONS_NONE, &pdc2));
133
134 /* create the simple 2-buffer swap chain */
135
136 DXGI_SWAP_CHAIN_DESC1 swapchaind = { 0 };
137 swapchaind.Format = DXGI_FORMAT_B8G8R8A8_UNORM;
138 swapchaind.SampleDesc.Count = 1;
139 swapchaind.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
140 swapchaind.BufferCount = 2;
141 swapchaind.Scaling = DXGI_SCALING_STRETCH;
142 swapchaind.SwapEffect = DXGI_SWAP_EFFECT_FLIP_SEQUENTIAL;
143 ThrowError(pfactxgi->CreateSwapChainForHwnd(pdev3.Get(),
144 iwapp.hwnd,
145 &swapchaind,
146 nullptr, nullptr,
147 &pswapchain));
148
149 /* create the back buffer bitmap for the swap chain and install it in the
150 device context */
151
152 CreateBuffer(pdc2, pbmpBackBuf);
153 pdc2->SetTarget(pbmpBackBuf.Get());
154 fDirty = true;
155
156 RTC::RebuildRegisteredDevDeps(iwapp);
157}

163void RTCFLIP::Present(com_ptr<ID2D1DeviceContext>& pdc2, const RC& rcgUpdate)
164{
165 if (rcgUpdate.fEmpty())
166 return;
167 DXGI_PRESENT_PARAMETERS pp = { 0 };
168 RECT rectUpdate = rcgUpdate;
169 RECT rectClient;
170 GetClientRect(iwapp.hwnd, &rectClient);
171 if (rectUpdate.left > rectClient.left || rectUpdate.top > rectClient.top ||
172 rectUpdate.right < rectClient.right || rectUpdate.bottom < rectClient.bottom) {
173 assert(!fDirty);
174 pp.DirtyRectsCount = 1;
175 pp.pDirtyRects = &rectUpdate;
176 }
177 else
178 fDirty = false;
179
180 HRESULT err = pswapchain->Present1(1, 0, &pp);
181}

183void RTCFLIP::CreateBuffer(com_ptr<ID2D1DeviceContext>& pdc2, com_ptr<ID2D1Bitmap1>& pbmpBuf)
184{
185 com_ptr<IDXGISurface> psurfdxgi;
186 ThrowError(pswapchain->GetBuffer(0, __uuidof(IDXGISurface), &psurfdxgi));
187 DXGI_SURFACE_DESC surfdesc;
188 psurfdxgi->GetDesc(&surfdesc);
189 float dxy = (float)::GetDpiForWindow(iwapp.hwnd);
190
191 D2D1_BITMAP_PROPERTIES1 bmpprop = {
192 PixelFormat(DXGI_FORMAT_B8G8R8A8_UNORM, D2D1_ALPHA_MODE_IGNORE),
193 dxy, dxy,
194 D2D1_BITMAP_OPTIONS_TARGET | D2D1_BITMAP_OPTIONS_CANNOT_DRAW };
195 ThrowError(pdc2->CreateBitmapFromDxgiSurface(psurfdxgi.Get(), &bmpprop, &pbmpBuf));
196}
