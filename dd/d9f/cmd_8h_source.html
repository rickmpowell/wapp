3/*
4 * cmd.h
5 *
6 * We provide a mechanism for generating and executing user interface
7 * commands. Commands correspond to the operations initiated by menus or
8 * buttons, but can be generalized to come from just about any source.
9 *
10 * The process we encourage is to instantiate a command object, which
11 * derives from the ICMD interface, and register it with a UI element.
12 * When the UI element triggers a event, the cmmand will be invoked by
13 * passing it into IWAPP::FExecuteCmd, which, in the simplest applications,
14 * will simply execute the command. FOr applications that support more
15 * advanced UI features, FExecuteCmd can be overridden to implement things
16 * like a command recorder, or an undo stack.
17 *
18 * Command objects also include standard interfaces to provide enable and
19 * disable state, menu text, and tooltip text.
20 */

124class menuiterator
125{
126public:
127 using iterator_category = input_iterator_tag;
128 using value_type = MENUITEMINFOW;
129 using difference_type = ptrdiff_t;
130 using pointer = MENUITEMINFOW*;
131 using reference = MENUITEMINFO&;
132
133 menuiterator(HMENU hmenu, int pos) : hmenu(hmenu), pos(pos)
134 {
135 memset(&mii, 0, sizeof(mii));
136 }
137
138 menuiterator& operator++()
139 {
140 ++pos;
141 mii.cbSize = 0; // invalidate the mii
142 return *this;
143 }
144
145 menuiterator operator++ (int)
146 {
147 if (mii.cbSize == 0)
148 UpdateMmi();
149 menuiterator it = *this;
150 mii.cbSize = 0;
151 pos++;
152 return it;
153 }
154
155 bool operator == (const menuiterator& it) const
156 {
157 return hmenu == it.hmenu && pos == it.pos;
158 }
159
160 bool operator != (const menuiterator& it) const
161 {
162 return !(*this == it);
163 }
164
165 difference_type operator - (const menuiterator& it) const
166 {
167 return static_cast<difference_type>(pos) - static_cast<difference_type>(it.pos);
168 }
169
170 reference operator * ()
171 {
172 if (mii.cbSize == 0)
173 UpdateMmi();
174 return mii;
175 }
176
177 pointer operator -> ()
178 {
179 if (mii.cbSize == 0)
180 UpdateMmi();
181 return &mii;
182 }
183
184 void UpdateMmi(void)
185 {
186 mii.cbSize = sizeof(mii);
187 mii.fMask = MIIM_ID | MIIM_FTYPE | MIIM_SUBMENU;
188 mii.cch = 0;
189 mii.dwTypeData = NULL;
190 if (!::GetMenuItemInfoW(hmenu, (UINT)pos, TRUE, &mii))
191 hmenu = NULL;
192 }
193
194private:
195 HMENU hmenu;
196 MENUITEMINFOW mii;
197 int pos;
198};
