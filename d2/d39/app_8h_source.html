211class resource_ptr
212{
213public:
214 resource_ptr(APP& app, string_view sType, unsigned rs) : hData(NULL), pData(nullptr)
215 {
216 HRSRC hrsrc = ::FindResourceW(app.hinst, MAKEINTRESOURCEW(rs), WsFromS(sType).c_str());
217 if (hrsrc == NULL)
218 throw ERRLAST();
219 cbData = static_cast<unsigned>(::SizeofResource(app.hinst, hrsrc));
220 hData = ::LoadResource(app.hinst, hrsrc);
221 if (hData == NULL)
222 throw ERRLAST();
223 pData = static_cast<BYTE*>(::LockResource(hData));
224 if (pData == NULL) {
225 hData = NULL;
226 throw ERRLAST();
227 }
228 }
229
230 // don't do reset on these move operations so cbData copies too
231
232 resource_ptr(resource_ptr&& ptr) noexcept :
233 hData(ptr.hData),
234 pData(ptr.pData),
235 cbData(ptr.cbData)
236 {
237 ptr.release();
238 }
239
240 resource_ptr& operator = (resource_ptr&& ptr) noexcept
241 {
242 if (this != &ptr) {
243 hData = ptr.hData;
244 pData = ptr.pData;
245 cbData = ptr.cbData;
246 ptr.release();
247 }
248 return *this;
249 }
250
251 ~resource_ptr() noexcept
252 {
253 reset();
254 }
255
256 resource_ptr(const resource_ptr&) = delete;
257 resource_ptr& operator = (const resource_ptr&) = delete;
258
259 HGLOBAL release(void) noexcept
260 {
261 /* we don't need to unlock resources */
262 HGLOBAL hT = hData;
263 hData = NULL;
264 pData = nullptr;
265 cbData = 0;
266 return hT;
267 }
268
269 void reset(HGLOBAL hDataNew = NULL) noexcept
270 {
271 hData = hDataNew;
272 if (hData) {
273 pData = static_cast<BYTE*>(::LockResource(hData));
274 assert(pData);
275 cbData = 0; /* note that we lose the size on reset because we don't have access to the hinst */
276 }
277 else {
278 pData = nullptr;
279 cbData = 0;
280 }
281 }
282
283 void swap(resource_ptr& ptr) noexcept
284 {
285 std::swap(hData, ptr.hData);
286 std::swap(pData, ptr.pData);
287 std::swap(cbData, ptr.cbData);
288 }
289

292 return pData;
293 }
294

296 {
297 return hData;
298 }
299
